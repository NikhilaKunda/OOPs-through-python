class Canara:
    def __init__(self):
        self.balance=0
        print("Welcome to the Canara Bank")
    def deposit(self):
        amount=float(input("Enter amount to deposited:"))
        self.balance+=amount
        print("\n deposited amount is:",amount)
    def withdraw(self):
        amount=float(input("Enter amount to withdraw:"))
        if self.balance>=amount:
          self.balance-=amount
          print("\n withdraw amount is:",amount)
        else:
            print("\n insufficient balance")
    def checkbalance(self):
        print("\n net available balance is:",self.balance)
#driver code
obj1=Canara()
obj1.deposit()
obj1.withdraw()
obj1.checkbalance()

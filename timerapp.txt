import tkinter as nikki

class TimerApp:
     def __init__(self,root):
       self.root=root
       self.root.title("Timer App")
       #initialize timer variable
       self.seconds=0
       self.timer_tiktik = False
       #create a label to display the timer
       self.timer_label=nikki.Label(root,text="00:00:00",font=("Helvetica",48))
       self.timer_label.pack(pady=20)
       #self.start and stop buttons
       self.start_button=nikki.Button(root,text="Start",command=self.start_timer)
       self.stop_button=nikki.Button(root,text="Stop",command=self.stop_timer)
       self.start_button.pack(side=nikki.LEFT,padx=10)
       self.stop_button.pack(side=nikki.LEFT,padx=10)
       #update timer display
       self.update_timer()

     def start_timer(self):
          if not self.timer_tiktik:
                    self.timer_tiktik = True
                    self.update_timer()
     def stop_timer(self):
               self.timer_tiktik = False
     def update_timer(self):
               if self.timer_tiktik:
                         self.seconds += 1
                         hours=self.seconds // 3600
                         minutes =( self.seconds %3600)// 60
                         seconds = self.seconds % 60
                         time_str = f" {hours:02}:{minutes:02}:{seconds:02}"
                         self.timer_label.config(text=time_str)
                         self.root.after(1000, self.update_timer)  # Update every 1 second

if __name__=="__main__":
          root = nikki.Tk()
          app = TimerApp(root)
          root.mainloop()
